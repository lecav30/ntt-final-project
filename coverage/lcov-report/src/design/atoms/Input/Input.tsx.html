import type { FC } from 'react';
import {
  Pressable,
  TextInput,
  View,
  type StyleProp,
  type TextInputProps,
  type ViewStyle,
} from 'react-native';
import Svg, { Path } from 'react-native-svg';
import styles from './Input.styles';
import useInput from './useInput';
 
interface Props {
  style?: StyleProp<ViewStyle>;
  clearButtonTestID?: string;
}
 
const Input: FC<TextInputProps & Props> = ({
  style,
  testID,
  clearButtonTestID,
  ...props
}) => {
  const {
    value: externalValue,
    onChangeText: externalOnChangeText,
    ...rest
  } = props;
 
  const { value: internalValue, setValue: setInternalValue } = useInput(
    externalValue ?? ''
  );
 
  const isControlled = externalValue !== undefined;
 
  const value = isControlled ? externalValue : internalValue;
 
  const handleChange = (text: string) => {
    if (isControlled && externalOnChangeText) {
      externalOnChangeText(text);
    } else {
      setInternalValue(text);
    }
  };
 
  const handleClear = () => {
    if (isControlled && externalOnChangeText) {
      externalOnChangeText('');
    } else {
      setInternalValue('');
    }
  };
 
  return (
    <View style={[styles.container, style]}>
      <TextInput
        {...rest}
        style={styles.input}
        value={value}
        onChangeText={handleChange}
        testID={testID}
      />
 
      {value.length > 0 && (
        <Pressable
          style={styles.clearButton}
          onPress={handleClear}
          testID={clearButtonTestID}
        >
          <Svg width="20" height="20" viewBox="0 0 14 14" fill="none">
